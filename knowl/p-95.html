\begin{equation*} \dx(uv){}=u\dx{v}{}+v\dx{u}{.} \end{equation*}
Rearranging this a bit we have
\begin{equation*} u\dx{v}{}=\dx(uv){}-v\dx{u}. \end{equation*}
or
\begin{equation*} \int{}u\dx{v}{}=\int{} \left(\dx(uv){}-v\dx{u}{}\right) = \int\dx(uv){}-\int v\dx{u}{=uv-\int v\dx{u}{}}. \end{equation*}
